def string_categories(survival):
    if survival == 0:
        return 'Died'
    else:
        return 'Survived'

titanic_df_drop_na['survived'] = titanic_df_drop_na['survived'].apply(string_categories)

sns.countplot(
    data=titanic_df_drop_na,
    x='survived',

    palette='seismic'
